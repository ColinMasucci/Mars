Program            → { Import } [ Component ] ;

Import             → "import" IDENTIFIER ";" ;

Component          → "component" IDENTIFIER [ "extends" IDENTIFIER ]
                      "{"
                        Subcomponents
                        Params
                        Functions
                      "}" ;

Subcomponents      → "subcomponents" "{" SubcomponentList "}" ;  // can make this more compact (1 line)
SubcomponentList   → { Subcomponent } ;
Subcomponent       → IDENTIFIER IDENTIFIER ";" ;

Params             → "params" "{" ParamList "}" ;
ParamList          → { Param } ;
Param              → Type IDENTIFIER ";" ;

Functions          → "functions" "{" FunctionList "}" ;
FunctionList       → { Function } ;
Function           → Type IDENTIFIER "(" [ ArgumentList ] ")" Block ;

ArgumentList      → Argument { "," Argument } ;
Argument          → Type IDENTIFIER ;

Type               → PrimitiveType { "[" "]" } ;
PrimitiveType      → "float" | "int" | "string" | "bool" | "void" ; //fix: void cannot be an array -> handle in parser not grammar

Block              → "{" [ StatementList ] "}" ;

StatementList      → [StatementList] Statement ;    // need statement list somewhere    

Statement          → Block
                   | IfStmt
                   | WhileStmt
                   | ForStmt
                   | ReturnStmt
                   | DeclStmt
                   | AssignStmt
                   | CallStmt
                   ;


DeclStmt           → Type IDENTIFIER [ "=" Expression ] ";" ;

IfStmt             → "if" "(" Expression ")" Statement [ "else" Statement ] ;
WhileStmt          → "while" "(" Expression ")" Statement ;

//fix: come back to how this works
ForStmt            → "for" "(" [ ExprOrDeclOrEmpty ] ";" [ Expression ] ";" [ Expression ] ")" Statement ;
ExprOrDeclOrEmpty  → ExprStmt | DeclStmt | /* empty */ ;

ReturnStmt         → "return" [ Expression ] ";" ;

AssignStmt         → IDENTIFIER AssignOp Expression ";" ;
AssignOp           → "=" | "+=" | "-=" | "*=" | "/=" ;


Expression         → Expression BinOp Expression | UnOpPre Expression | "abs(" Expression ")" | UnOpPost Expression | "(" Expression ")" | Literal ; 

BinOp              → MathOp | CompOp | LogicOp ; 
MathOp             → "+" | "-" | "/" | "*" | "^" ; 
CompOp             → "==" | "<=" | ">=" | ">" | "<" | "!=" ; 
Logic              → "&&" | "||" ; 

UnOpPre            → "!" | "-" ;
UnOpPost           → "++" | "--" ;

Literal            → INT | FLOAT | STRING | IDENTIFIER { "[" INT "]" } | "true" | "false" ;
